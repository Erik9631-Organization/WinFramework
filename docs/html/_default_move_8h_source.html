110 template<class T>
111 void DefaultMove<T>::CalculateAbsolutePosition()
112 {
113  if (associatedAdjustableNode.IsRoot() || associatedAdjustableNode.GetParent()->IsRoot()) //If the parent is root, we are in the global windowSpace and relative is same as absolute
114  {
115  absolutePosition = relativePosition + position;
116  }
117  else
118  {
119  absolutePosition.X = relativePosition.X + associatedAdjustableNode.GetParent()->GetValue().GetAbsoluteX() + position.X;
120  absolutePosition.Y = relativePosition.Y + associatedAdjustableNode.GetParent()->GetValue().GetAbsoluteY() + position.Y;
121  }
122 }
